222 def showTable(self, table, rowSeparatorChar = "-", colSeparatorChar = "|"):
223 """@brief Show the contents of a table to the user.
224 @param table This must be a list. Each list element must be a table row (list).
225 Each element in each row must be a string.
226 @param rowSeparatorChar The character used for horizontal lines to separate table rows.
227 @param colSeparatorChar The character used to separate table columns."""
228 columnWidths = []
229 # Check we have a table to display
230 if len(table) == 0:
231 raise Exception("No table rows to display")
232
233 # Check all rows have the same number of columns in the table
234 colCount = len(table[0])
235 for row in table:
236 if len(row) != colCount:
237 raise Exception(f"{str(row)} column count different from first row ({colCount})")
238
239 for row in table:
240 for col in row:
241 if not isinstance(col, str):
242 raise Exception(f"Table column is not a string: {col} in {row}")
243
244 # Get the max width for each column
245 for col in range(0,colCount):
246 maxWidth=0
247 for row in table:
248 if len(row[col]) > maxWidth:
249 maxWidth = len(row[col])
250 columnWidths.append(maxWidth)
251
252 tableWidth = 1
253 for columnWidth in columnWidths:
254 tableWidth += columnWidth + 3 # Space each side of the column + a column divider character
255
256 # Add the top line of the table
257 self.info(rowSeparatorChar*tableWidth)
258
259 # The starting row index
260 for rowIndex in range(0, len(table)):
261 rowText = colSeparatorChar
262 colIndex = 0
263 for col in table[rowIndex]:
264 colWidth = columnWidths[colIndex]
265 rowText = rowText + " " + f"{col:>{colWidth}s}" + " " + colSeparatorChar
266 colIndex += 1
267 self.info(rowText)
268 # Add the row separator line
269 self.info(rowSeparatorChar*tableWidth)

451 def _startProgress(self, durationSeconds=0, startMessage=None, expectedMsgList=[], expectedMsgCount=0):
452 """@brief Start a timer that will update the progress bar.
453 The progress bar can simply update on a timer every second with durationSeconds set to the expected length
454 of the task.
455
456 If startMessage is set to a text string the progress time will not start until the log message area contains
457 the start message.
458
459 Alternatively if expectedMsgList contains a list of strings we expect to receive then the progress bar is
460 updated as each message is received. The messages may be the entire line of a log message or parts of a
461 log message line.
462
463 Alternatively if expectedMsgCount is set to a value > 0 then the progress bar is updated as each message is
464 added to the log and reaches 100% when the number of messages added to the log file reaches the expectedMsgCount.
465
466 @param startMessage The text of the log message we expect to receive to trigger the progress bar timer start.
467 @param expectedMsgList A list of the expected log file messages.
468 @param expectedMsgCount A int value that defines the number of log messages we expect to receive for normal progress
469 completion."""
470 self._progressValue = 0
471 self._progressBarStartMessage = ""
472 self._progressBarExpectedMessageList = []
473 self._expectedProgressBarMessageIndex = 0
474 self._expectedProgressBarMsgCount = 0
475 self._updateProgressOnTimer = False
476 self._progress.set_value( self._progressValue )
477 # If the caller wants to the progress bar to update as the log file message count increases.
478 if expectedMsgCount > 0:
479 self._expectedProgressBarMsgCount = expectedMsgCount
480 self._progressStepValue = TabbedNiceGui.MAX_PROGRESS_VALUE/float(self._expectedProgressBarMsgCount)
481
482 # If the caller wants to update the progress bar on expected messages.
483 elif len(expectedMsgList):
484 #Use the text of log messages to increment the progress bar.
485 self._expectedProgressBarMessageIndex = 0
486 self._progressBarExpectedMessageList = expectedMsgList
487 self._progressStepValue = TabbedNiceGui.MAX_PROGRESS_VALUE/float(len(expectedMsgList))
488
489 elif durationSeconds > 0:
490 # Calc the step size required to give the required duration
491 self._progressStepValue = TabbedNiceGui.MAX_PROGRESS_VALUE/float(durationSeconds)
492 if startMessage:
493 self._progressBarStartMessage = startMessage
494 else:
495 # Start updating the progress bar now.
496 self._updateProgressOnTimer = True
497
498 else:
499 raise Exception("BUG: _startProgressTimer() called. len(expectedMsgList)=0 and durationSeconds<=0.")
500
501 self._progress.set_visibility(True)

508 def _updateProgressBar(self, msg):
509 """@brief Update the progress bar if required when a log message is received. This is called as each message is added to the log.
510 @param msg The log message received."""
511 # If we update the progress bar as each message is received until we have a log with self._expectedProgressBarMsgCount many messages.
512 if self._expectedProgressBarMsgCount > 0:
513 self._progressValue = self._progressValue + self._progressStepValue
514 self._progress.set_value( self._progressValue )
515
516 # If we have a list of log messages to update the progress bar.
517 elif len(self._progressBarExpectedMessageList) > 0:
518 if self._expectedProgressBarMessageIndex < len(self._progressBarExpectedMessageList):
519 # Get the message we expect to receive next
520 expectedMsg = self._progressBarExpectedMessageList[self._expectedProgressBarMessageIndex]
521 if msg.find(expectedMsg) != -1:
522 self._progressValue = self._progressValue + self._progressStepValue
523 self._progress.set_value( self._progressValue )
524 self._expectedProgressBarMessageIndex += 1
525
526 # If we have a message that we expect to receive to start the progress bar timer.
527 elif self._progressBarStartMessage and len(self._progressBarStartMessage):
528 # If we found the start message in the message received.
529 if msg.find(self._progressBarStartMessage) != -1:
530 # Start updating the progress bar now on the timer.
531 self._updateProgressOnTimer = True

608 def _updateGUIAndWaitForResponse(self, msgDict, timeout=DEFAULT_GUI_RESPONSE_TIMEOUT):
609 """@brief Send a message to the GUI and wait for a response.
610 @param msgDict The message dictionary to be sent to the GUI.
611 @param timeout The number of seconds to wait for a response.
612 @return The return dict."""
613 timeoutT = time()+timeout
614 rxDict = None
615 self._updateGUI(msgDict)
616 while True:
617 if not self._fromGUIQueue.empty():
618 rxMessage = self._fromGUIQueue.get()
619 if isinstance(rxMessage, dict):
620 rxDict = rxMessage
621 break
622
623 elif time() >= timeoutT:
624 raise Exception(f"{timeout} second GUI response timeout.")
625
626 else:
627 # Don't spin to fast
628 sleep(0.1)
629
630 return rxDict
