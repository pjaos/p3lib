222 def showTable(self, table, rowSeparatorChar = "-", colSeparatorChar = "|"):
223 """@brief Show the contents of a table to the user.
224 @param table This must be a list. Each list element must be a table row (list).
225 Each element in each row must be a string.
226 @param rowSeparatorChar The character used for horizontal lines to separate table rows.
227 @param colSeparatorChar The character used to separate table columns."""
228 columnWidths = []
229 # Check we have a table to display
230 if len(table) == 0:
231 raise Exception("No table rows to display")
232
233 # Check all rows have the same number of columns in the table
234 colCount = len(table[0])
235 for row in table:
236 if len(row) != colCount:
237 raise Exception(f"{str(row)} column count different from first row ({colCount})")
238
239 for row in table:
240 for col in row:
241 if not isinstance(col, str):
242 raise Exception(f"Table column is not a string: {col} in {row}")
243
244 # Get the max width for each column
245 for col in range(0,colCount):
246 maxWidth=0
247 for row in table:
248 if len(row[col]) > maxWidth:
249 maxWidth = len(row[col])
250 columnWidths.append(maxWidth)
251
252 tableWidth = 1
253 for columnWidth in columnWidths:
254 tableWidth += columnWidth + 3 # Space each side of the column + a column divider character
255
256 # Add the top line of the table
257 self.info(rowSeparatorChar*tableWidth)
258
259 # The starting row index
260 for rowIndex in range(0, len(table)):
261 rowText = colSeparatorChar
262 colIndex = 0
263 for col in table[rowIndex]:
264 colWidth = columnWidths[colIndex]
265 rowText = rowText + " " + f"{col:>{colWidth}s}" + " " + colSeparatorChar
266 colIndex += 1
267 self.info(rowText)
268 # Add the row separator line
269 self.info(rowSeparatorChar*tableWidth)

428 def _startProgress(self, durationSeconds=0, startMessage=None, expectedMsgList=[], expectedMsgCount=0):
429 """@brief Start a timer that will update the progress bar.
430 The progress bar can simply update on a timer every second with durationSeconds set to the expected length
431 of the task.
432
433 If startMessage is set to a text string the progress time will not start until the log message area contains
434 the start message.
435
436 Alternatively if expectedMsgList contains a list of strings we expect to receive then the progress bar is
437 updated as each message is received. The messages may be the entire line of a log message or parts of a
438 log message line.
439
440 Alternatively if expectedMsgCount is set to a value > 0 then the progress bar is updated as each message is
441 added to the log and reaches 100% when the number of messages added to the log file reaches the expectedMsgCount.
442
443 @param startMessage The text of the log message we expect to receive to trigger the progress bar timer start.
444 @param expectedMsgList A list of the expected log file messages.
445 @param expectedMsgCount A int value that defines the number of log messages we expect to receive for normal progress
446 completion."""
447 self._progressValue = 0
448 self._progressBarStartMessage = ""
449 self._progressBarExpectedMessageList = []
450 self._expectedProgressBarMessageIndex = 0
451 self._expectedProgressBarMsgCount = 0
452 self._updateProgressOnTimer = False
453 self._progress.set_value( self._progressValue )
454 # If the caller wants to the progress bar to update as the log file message count increases.
455 if expectedMsgCount > 0:
456 self._expectedProgressBarMsgCount = expectedMsgCount
457 self._progressStepValue = TabbedNiceGui.MAX_PROGRESS_VALUE/float(self._expectedProgressBarMsgCount)
458
459 # If the caller wants to update the progress bar on expected messages.
460 elif len(expectedMsgList):
461 #Use the text of log messages to increment the progress bar.
462 self._expectedProgressBarMessageIndex = 0
463 self._progressBarExpectedMessageList = expectedMsgList
464 self._progressStepValue = TabbedNiceGui.MAX_PROGRESS_VALUE/float(len(expectedMsgList))
465
466 elif durationSeconds > 0:
467 # Calc the step size required to give the required duration
468 self._progressStepValue = TabbedNiceGui.MAX_PROGRESS_VALUE/float(durationSeconds)
469 if startMessage:
470 self._progressBarStartMessage = startMessage
471 else:
472 # Start updating the progress bar now.
473 self._updateProgressOnTimer = True
474
475 else:
476 raise Exception("BUG: _startProgressTimer() called. len(expectedMsgList)=0 and durationSeconds<=0.")
477
478 self._progress.set_visibility(True)

485 def _updateProgressBar(self, msg):
486 """@brief Update the progress bar if required when a log message is received. This is called as each message is added to the log.
487 @param msg The log message received."""
488 # If we update the progress bar as each message is received until we have a log with self._expectedProgressBarMsgCount many messages.
489 if self._expectedProgressBarMsgCount > 0:
490 self._progressValue = self._progressValue + self._progressStepValue
491 self._progress.set_value( self._progressValue )
492
493 # If we have a list of log messages to update the progress bar.
494 elif len(self._progressBarExpectedMessageList) > 0:
495 if self._expectedProgressBarMessageIndex < len(self._progressBarExpectedMessageList):
496 # Get the message we expect to receive next
497 expectedMsg = self._progressBarExpectedMessageList[self._expectedProgressBarMessageIndex]
498 if msg.find(expectedMsg) != -1:
499 self._progressValue = self._progressValue + self._progressStepValue
500 self._progress.set_value( self._progressValue )
501 self._expectedProgressBarMessageIndex += 1
502
503 # If we have a message that we expect to receive to start the progress bar timer.
504 elif self._progressBarStartMessage and len(self._progressBarStartMessage):
505 # If we found the start message in the message received.
506 if msg.find(self._progressBarStartMessage) != -1:
507 # Start updating the progress bar now on the timer.
508 self._updateProgressOnTimer = True

585 def _updateGUIAndWaitForResponse(self, msgDict, timeout=DEFAULT_GUI_RESPONSE_TIMEOUT):
586 """@brief Send a message to the GUI and wait for a response.
587 @param msgDict The message dictionary to be sent to the GUI.
588 @param timeout The number of seconds to wait for a response.
589 @return The return dict."""
590 timeoutT = time()+timeout
591 rxDict = None
592 self._updateGUI(msgDict)
593 while True:
594 if not self._fromGUIQueue.empty():
595 rxMessage = self._fromGUIQueue.get()
596 if isinstance(rxMessage, dict):
597 rxDict = rxMessage
598 break
599
600 elif time() >= timeoutT:
601 raise Exception(f"{timeout} second GUI response timeout.")
602
603 else:
604 # Don't spin to fast
605 sleep(0.1)
606
607 return rxDict
