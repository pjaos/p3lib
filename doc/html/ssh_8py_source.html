51 def __exec_command(self, command, bufsize=-1, timeout=None):
52 """
53 Execute a command on the SSH server. A new L{Channel} is opened and
54 the requested command is executed. The command's input and output
55 streams are returned as python C{file}-like objects representing
56 stdin, stdout, and stderr.
57
58 @param command: the command to execute
59 @type command: str
60 @param bufsize: interpreted the same way as by the built-in C{file()} function in python
61 @type bufsize: int
62 @param timeout The timeout value in seconds or None if no command timeout.
63 @param type None or float
64 @return: the stdin, stdout, and stderr of the executing command
65 @rtype: tuple(L{ChannelFile}, L{ChannelFile}, L{ChannelFile})
66
67 @raise SSHException: if the server fails to execute the command
68 """
69 chan = self._transport.open_session(timeout=timeout)
70 chan.settimeout(timeout)
71 chan.exec_command(command)
72 stdin = chan.makefile('wb', bufsize)
73 stdout = chan.makefile('rb', bufsize)
74 stderr = chan.makefile_stderr('rb', bufsize)
75 exitStatus = chan.recv_exit_status()
76 return stdin, stdout, stderr, exitStatus

92 def runCmd(self, cmd, throwError=True, timeout=None):
93 """@brief Run a command over an ssh session
94 @param cmd The command to execute
95 @param throwError If True and the command fails throw an error. If False and the command fail return the erorr code etc.
96 If throwError is True and the exit status of the command is not 0 then an SSHError will be thrown
97 @param timeout The timeout value in seconds or None (default) if no command timeout is required.
98 @return A tuple with the following three elements if the command executes
99 0 - the return code/exit status of the command
100 1 - Lines of text from stdout
101 2 - lines of text from stderr
102 If the command fails to execute then None is returned.
103 """
104 try:
105 stdin, stdout, stderr, exitStatus = self.__exec_command(cmd,timeout=timeout)
106 if throwError and exitStatus != 0:
107 errorText = stderr.read()
108 if len(errorText) > 0:
109 raise SSHError(errorText)
110 raise SSHError("The cmd '%s' return the error code: %d" % (cmd, exitStatus))
111 return [exitStatus, ExtendedSSHClient.GetLines(stdout.read()), ExtendedSSHClient.GetLines(stderr.read())]
112 except:
113 if throwError:
114 raise
115 return None

556 def getAuthKeyBackupFile(self, maxBackupFileCount=10):
557 """@brief Get the name of the backup name for the authorised keys file.
558 @param maxBackupFileCount The maximum number of backup files to keep.
559 @return None
560 - We create up to maxBackupFileCount backup files.
561 - Once all the backup files have been created we always replace the oldest
562 backup file.
563 - The backup files have the suffix .backup1, .backup2 etc.
564 """
565 authKeysFile = self.getRemoteAuthorisedKeyFile()
566 authKeyBackupFilePart = "%s.backup" % (authKeysFile)
567 suffixNum = 1
568 while True:
569 authKeyBackupFileName = "%s%d" % (authKeyBackupFilePart, suffixNum)
570 rc, stdoutLines, stderrLines = self.runCmd("test -f %s" % (authKeyBackupFileName), throwError=False)
571 if rc != 0:
572 return authKeyBackupFileName
573 # If all the backup files have been created
574 if suffixNum >= maxBackupFileCount:
575 # List the files in creation order (oldest first)
576 rc, stdoutLines, stderrLines = self.runCmd("ls -ltr %s*" % (authKeyBackupFilePart), throwError=False)
577 if rc == 0:
578 if len(stdoutLines) > 0:
579 elems = stdoutLines[0].split()
580 if len(elems) > 0:
581 # Return the oldest file as the next backup filename so that we roll
582 # around the always replacing the oldest backup file.
583 backupfile = elems[len(elems) - 1]
584 return backupfile
585
586 raise SSHError("Unable to %s to %s. Please manually remove the backup files on the ssh server." % (
587 authKeysFile, authKeyBackupFilePart))
588
589 suffixNum = suffixNum + 1

620 self.runCmd("rm -f %s" % (tmpAuthKeysFile), throwError=False)
621

623 self.runCmd("touch %s" % (tmpAuthKeysFile))
624

630

632 self.runCmd("mv %s %s" % (authKeysFile, previousAuthKeysFile))

636 self._info("{} public key not found in ssh server authorised keys file.".format(publicKeyID))
