243 def _create_gnome_desktop_files(self):
244 """@brief Create the gnome desktop file for this app."""

423 def _convert_png_to_icns(self):
424 """@brief Convert the iniital png file to an isnc file for use on MacOS.
425 This method can only be called on MacOS
426 @return The abs path of the converted icns file."""
427 base = Image.open(self._abs_icon_file)
428 sizes = [16, 32, 128, 256, 512, 1024]
429
430 icon_folder = os.path.dirname(self._abs_icon_file)
431 iconset = Path(icon_folder, 'my.iconset')
432 iconset.mkdir(exist_ok=True)
433
434 for size in sizes:
435 img = base.resize((size, size))
436 filename = iconset / f'icon_{size}x{size}.png'
437 img.save(filename)
438
439 # use iconutil MacOS util program to create the icns files
440 subprocess.run(['iconutil', '--convert', 'icns', iconset])
441
442 # Clean the png files created
443 if iconset.exists():
444 shutil.rmtree(iconset)
445
446 return str(iconset).replace('.iconset', '.icns')

465 def _create_app(self):
466 """@brief Create a MacOS app folder with the required files to launch an app."""
467 self.deletedeletedeletedelete()

474 python_path = self._get_python_path()

476 exec_path.write_text(f"#!/bin/bash\nexec {python_path} -m {main_module}\n")
477 exec_path.chmod(0x755)

480 plist = {
481 'CFBundleName': self._app_name_app_name,
482 'CFBundleIdentifier': f'com.example.{self._app_name.lower()}',

484 'CFBundlePackageType': 'APPL',
