61 if not icon_file.endswith('.png'):

368 def _convert_png_to_icns(self):
369 """@brief Convert the iniital png file to an isnc file for use on MacOS.
370 This method can only be called on MacOS
371 @return The abs path of the converted icns file."""
372 base = Image.open(self._abs_icon_file)
373 sizes = [16, 32, 128, 256, 512, 1024]
374
375 icon_folder = os.path.dirname(self._abs_icon_file)
376 iconset = Path(icon_folder, 'my.iconset')
377 iconset.mkdir(exist_ok=True)
378
379 for size in sizes:
380 img = base.resize((size, size))
381 filename = iconset / f'icon_{size}x{size}.png'
382 img.save(filename)
383
384 # use iconutil MacOS util program to create the icns files
385 subprocess.run(['iconutil', '--convert', 'icns', '--output', 'my.icns', iconset])
386
387 # Clean the png files created
388 if iconset.exists():
389 shutil.rmtree(iconset)
390
391 return str(iconset).replace('.iconset', '.icns')
