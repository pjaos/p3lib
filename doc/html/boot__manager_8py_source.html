278 def add(self, user, argString=None, enableSyslog=False):
279 """@brief Add an executable file to the processes started at boot time.
280 This will also start the process. The executable file must be
281 named the same as the python file executed without the .py suffix.
282 @param user The Linux user that will run the executable file.
283 This should not be root as config files will be be saved
284 to non root user paths on Linux systems and the startup
285 script should then be executed with the same username in
286 order that the same config file is used.
287 If set to None then the current user is used.
288 @param argString The argument string that the program is to be launched with.
289 @param enableSyslog If True enable stdout and stderr to be sent to syslog."""
290 if user is None:
291 user = self._username
292
293 appName, absApp = self._getApp()
294
295 serviceFile = self._getServiceFile(appName)
296
297 lines = []
298 lines.append("[Unit]")
299 lines.append("After=network.target")
300 lines.append("StartLimitIntervalSec=0")
301 lines.append("")
302 lines.append("[Service]")
303 lines.append("Type=simple")
304 lines.append("Restart=always")
305 lines.append("RestartSec=1")
306 if enableSyslog:
307 lines.append("StandardOutput=syslog")
308 lines.append("StandardError=syslog")
309 else:
310 lines.append("StandardOutput=null")
311 lines.append("StandardError=journal")
312 if self._rootMode and user != 'root':
313 lines.append("User={}".format(user))
314
315 #We add the home path env var so that config files (if stored in/under
316 # the users home dir) can be found by the prgram.
317 if user and len(user) > 0:
318 lines.append('Environment="HOME=/home/{}"'.format(user))
319 if argString:
320 argString = argString.strip()
321 if argString.startswith(absApp):
322 argString=argString.replace(absApp, "")
323 # We don't want the enable cmd opt in the cmd we add to the systemd file.
324 if argString.find(BootManager.ENABLE_CMD_OPT):
325 argString = argString.replace(BootManager.ENABLE_CMD_OPT, "")
326 argString = argString.strip()
327 lines.append("ExecStart={} {}".format(absApp, argString))
328 else:
329 lines.append("ExecStart={}".format(absApp))
330 lines.append("")
331 lines.append("[Install]")
332 lines.append("WantedBy=multi-user.target")
333 lines.append("")
334
335 try:
336 fd = open(serviceFile, 'w')
337 fd.write( "\n".join(lines) )
338 fd.close()
339 self._info(f"Created {serviceFile}")
340 except IOError:
341 self._fatalError("Failed to create {}".format(serviceFile) )
342
343 cmd = "{} daemon-reload".format(self._cmdLinePrefix)
344 self._runLocalCmd(cmd)
345 cmd = "{} enable {}".format(self._cmdLinePrefix, appName)
346 self._info("Enabled {} on restart".format(appName))
347 self._runLocalCmd(cmd)
348 cmd = "{} start {}".format(self._cmdLinePrefix, appName)
349 self._runLocalCmd(cmd)
