357 def add(self, user, argString=None, enableSyslog=False):
358 """@brief Add an executable file to the processes started at boot time.
359 This will also start the process. The executable file must be
360 named the same as the python file executed without the .py suffix.
361 @param user The Linux user that will run the executable file.
362 This should not be root as config files will be be saved
363 to non root user paths on Linux systems and the startup
364 script should then be executed with the same username in
365 order that the same config file is used.
366 If set to None then the current user is used.
367 @param argString The argument string that the program is to be launched with.
368 @param enableSyslog If True enable stdout and stderr to be sent to syslog."""
369 if user is None:
370 user = self._username
371
372 appName, absApp = self._getApp()
373 serviceName = self._getServiceName()
374
375 serviceFile = self._getServiceFile(serviceName)
376
377 lines = []
378 lines.append("[Unit]")
379 lines.append("After=network.target")
380 lines.append("StartLimitIntervalSec=0")
381 lines.append("")
382 lines.append("[Service]")
383 lines.append("Type=simple")
384 lines.append("Restart=always")
385 lines.append(f"RestartSec={self._restartSeconds}")
386 if enableSyslog:
387 lines.append("StandardOutput=syslog")
388 lines.append("StandardError=syslog")
389 else:
390 lines.append("StandardOutput=null")
391 lines.append("StandardError=journal")
392 if self._rootMode and user != 'root':
393 lines.append("User={}".format(user))
394
395 #We add the home path env var so that config files (if stored in/under
396 # the users home dir) can be found by the prgram.
397 if user and len(user) > 0:
398 lines.append('Environment="HOME=/home/{}"'.format(user))
399 if argString:
400 argString = argString.strip()
401 if argString.startswith(absApp):
402 argString=argString.replace(absApp, "")
403 # We don't want the enable cmd opt in the cmd we add to the systemd file.
404 if argString.find(BootManager.ENABLE_CMD_OPT):
405 argString = argString.replace(BootManager.ENABLE_CMD_OPT, "")
406 argString = argString.strip()
407 absApp = self._prefix_with_venv_python(absApp)
408 lines.append("ExecStart={} {}".format(absApp, argString))
409 else:
410 lines.append("ExecStart={}".format(absApp))
411 lines.append("")
412 lines.append("[Install]")
413 lines.append("WantedBy=multi-user.target")
414 lines.append("")
415
416 try:
417 fd = open(serviceFile, 'w')
418 fd.write( "\n".join(lines) )
419 fd.close()
420 self._info(f"Created {serviceFile}")
421 except IOError:
422 self._fatalError("Failed to create {}".format(serviceFile) )
423
424 cmd = "{} daemon-reload".format(self._cmdLinePrefix)
425 self._runLocalCmd(cmd)
426 cmd = "{} enable {}".format(self._cmdLinePrefix, serviceName)
427 self._info("Enabled {} on restart".format(serviceName))
428 self._runLocalCmd(cmd)
429 cmd = "{} start {}".format(self._cmdLinePrefix, serviceName)
430 self._runLocalCmd(cmd)
431 self._info("Started {}".format(serviceName))

433 def _prefix_with_venv_python(self, python_file):
434 """@brief Given the python file to execute. Check if it sits in a venv folder and prefix it with the
435 venv python version if present. If we don't do this the wrong python binary maybe executed
436 which may be missing python modules in the venv.
437 @param python_file The python file to execute."""
438 new_python_file = python_file
439 venv_pos = python_file.find('venv')
440 if python_file.endswith(".py") and venv_pos > 0:
441 bin_path = os.path.join(python_file[:venv_pos+4], 'bin')
442 if os.path.isdir(bin_path):
443 for python_bin in ['python3', 'python']:
444 python_bin_file = os.path.join(bin_path, python_bin)
445 if os.path.isfile(python_bin_file):
446 new_python_file = python_bin_file + " " + python_file
447 break
448
449 return new_python_file
