123 def _getInstallledStartupScript(self):
124 """@brief Get the startup script full path. The startup script must be
125 named the same as the python file executed without the .py suffix.
126 @return The startup script file (absolute path)."""""
127 startupScript=None
128 pythonFile = sys.argv[0] # The python file executed at program startup
129 if pythonFile.startswith("./"):
130 pythonFile=pythonFile[2:]
131
132 envPaths = self._getPaths()
133
134 # Search first for a file that does not have the .py suffix as
135 # this may be installed via a deb/rpm installation.
136 exeFile=os.path.basename( pythonFile.replace(".py", "") )
137 if envPaths and len(envPaths) > 0:
138 for envPath in envPaths:
139 _exeStartupScript = os.path.join(envPath, exeFile)
140 if os.path.isfile(_exeStartupScript):
141 startupScript=_exeStartupScript
142 break
143
144 # If the script file has not been found then search for a file with
145 # the .py suffix.
146 if not startupScript:
147 if envPaths and len(envPaths) > 0:
148 for envPath in envPaths:
149 _startupScript = os.path.join(envPath, pythonFile)
150 if os.path.isfile(_startupScript):
151 startupScript=_startupScript
152 break
153
154 if not startupScript:
155 paths = self._getPaths()
156 if len(paths):
157 for _path in paths:
158 self._info(_path)
159 self._fatalError("{} startup script not found using the PATH env var".format(pythonFile) )
160 else:
161 self._fatalError("No PATH env var found.")
162
163 return startupScript

243 def add(self, user, argString=None, enableSyslog=False):
244 """@brief Add an executable file to the processes started at boot time.
245 This will also start the process. The executable file must be
246 named the same as the python file executed without the .py suffix.
247 @param user The Linux user that will run the executable file.
248 This should not be root as config files will be be saved
249 to non root user paths on Linux systems and the startup
250 script should then be executed with the same username in
251 order that the same config file is used.
252 If set to None then the current user is used.
253 @param argString The argument string that the program is to be launched with.
254 @param enableSyslog If True enable stdout and stderr to be sent to syslog."""

256 user = self._username

266 lines.append("")
267 lines.append("[Service]")
268 lines.append("Type=simple")
269 lines.append("Restart=always")
