14def initArgs(parser, lastCmdLineArg=None, checkHostArg=True):
15 """This method is responsible for
16 - Ensure that the host argument has been defined by the user on the command line
17 - Set the debug level as defined on the command line.
18 - If lastCmdLineArg is defined then ensure that this is the last arguemnt on the
19 command line. If we don't do this then the user may define some arguments on
20 the command line after the action (callback) command line option but as
21 the arguments are processed in sequence the args following the action
22 arg will not be used.
23 """

25 if checkHostArg:

27 raise Exception("Please define the RFeye host on the command line.")

29 parser.uio.debugLevel = parser.values.debug

31 # Check that the argument that invoked tha action (callback) is the last on the command line
32 argOk = False
33 if len(sys.argv) > 0:
34 lastArg = sys.argv[len(sys.argv) - 1]
35 if lastArg == lastCmdLineArg:
36 argOk = True
37

39 raise Exception("Please ensure %s (if used) is the last argument on the command line." % (lastCmdLineArg))

100def getDict(filename, jsonFmt=False):
101 """@brief Load dict from file
102 @param jsonFmt If True then we expect the file to be in json format.
103
104 if json is True we expect the file to be in json format
105
106 if json is False
107 We key=value pairs (= is the separate character).
108 Lines containing a hash sign as the first non-whitespace character are
109 ignored. Leading and trailing whitespace is ignored.
110
111 Lines not containing an equals sign are also silently ignored.
112
113 Lines not ignored are assumed to be in the form key=value, where key
114 does not contain an equals sign;
115 Control characters and double quotes in both key and value are silently
116 discarded. value is also truncated just before the first whitespace or
117 equals sign it contains.
118
119 @return Return the dict loaded from the file.
120
121 """
122 dictLoaded = {}
123
124 if jsonFmt:
125 fp = open(filename, 'r')
126 dictLoaded = json.load(fp)
127 fp.close()
128
129 else:
130
131 lines = getLinesFromFile(filename)
132 for line in lines:
133 # strip leading and trailing whitespaces
134 line = line.strip()
135 # if a comment line then ignore
136 if line.find('#') == 0:
137 continue
138 # add an entry to the dict
139 _addEntry(line, dictLoaded)
140
141 return dictLoaded

144def saveDict(dictToSave, filename, jsonFmt=False):
145 """@brief Save dict to a file.
146 @param jsonFmt If True then we expect the file to be in json format.
147
148 if json is True we expect the file to be in json format
149
150 if json is False the file is saved as key = value pairs
151 Each key in dict produces a line of the form key=value. Output will be
152 ambiguous if any keys contain equals signs or if any values contain
153 newlines.
154
155 """
156
157 if jsonFmt:
158 try:
159
160 with open(filename, "w") as write_file:
161 json.dump(dictToSave, write_file)
162
163 except IOError as i:
164 raise IOError(i.errno, 'Failed to write file \'%s\': %s'
165 % (filename, i.strerror), i.filename).with_traceback(sys.exc_info()[2])
166 else:
167 lines = []
168 # build config file lines
169 for key in list(dictToSave.keys()):
170 lines.append(str(key) + '=' + str(dictToSave.get(key)) + '\n')
171 try:
172 f = open(filename, 'w')
173 f.writelines(lines)
174 f.close()
175 except IOError as i:
176 raise IOError(i.errno, 'Failed to write file \'%s\': %s'
177 % (filename, i.strerror), i.filename).with_traceback(sys.exc_info()[2])
