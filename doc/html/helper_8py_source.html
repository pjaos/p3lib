125def getDict(filename, jsonFmt=False):
126 """@brief Load dict from file
127 @param jsonFmt If True then we expect the file to be in json format.
128
129 if json is True we expect the file to be in json format
130
131 if json is False
132 We key=value pairs (= is the separate character).
133 Lines containing a hash sign as the first non-whitespace character are
134 ignored. Leading and trailing whitespace is ignored.
135
136 Lines not containing an equals sign are also silently ignored.
137
138 Lines not ignored are assumed to be in the form key=value, where key
139 does not contain an equals sign;
140 Control characters and double quotes in both key and value are silently
141 discarded. value is also truncated just before the first whitespace or
142 equals sign it contains.
143
144 @return Return the dict loaded from the file.
145
146 """
147 dictLoaded = {}
148
149 if jsonFmt:
150 fp = open(filename, 'r')
151 dictLoaded = json.load(fp)
152 fp.close()
153
154 else:
155
156 lines = getLinesFromFile(filename)
157 for line in lines:
158 # strip leading and trailing whitespaces
159 line = line.strip()
160 # if a comment line then ignore
161 if line.find('#') == 0:
162 continue
163 # add an entry to the dict
164 _addEntry(line, dictLoaded)
165
166 return dictLoaded
