249  """@brief Save the text to the log file if one is defined.
250  @param text The text to be saved.
251  @param logFile The logFile to save data to.
252  @param addLF If True then a line feed is added to the output in the log file.
253  @param addDateTime If True add the date and time to the logfile.
254  @param symLinkFile The name of the fixed symlink file to point to the latest log file.
255  @return None"""
256  createSymLink = False
257  if logFile:
258  if addDateTime:
259  timeStr = datetime.now().strftime("%d/%m/%Y-%H:%M:%S.%f")
260  text = "{}: {}".format(strftime(timeStr, localtime()).lower(), text)
261 
262  # If the log file is about to be created then we will create a symlink
263  # to the file.
264  if not os.path.isfile(logFile):
265  createSymLink = True
266 
267  fd = open(logFile, 'a')
268  if addLF:
269  fd.write("{}\n".format(text))
270  else:
271  fd.write(text)
272  fd.close()

274  if createSymLink:
275  #This is helpful as the link will point to the latest log file
276  #which can be useful when debugging. I.E no need to find the
277  #name of the latest file.

279  # if the simlink has not been set then default to the logging file
280  if dirName is None:

283  if os.path.lexists(absSymLink):
