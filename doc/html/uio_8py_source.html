259  def _storeToLog(self, text, logFile, addLF=True, addDateTime=True, symLinkFile=USER_LOG_SYM_LINK):
260  """@brief Save the text to the log file if one is defined.
261  @param text The text to be saved.
262  @param logFile The logFile to save data to.
263  @param addLF If True then a line feed is added to the output in the log file.
264  @param symLinkFile The name of the fixed symlink file to point to the latest log file.
265  @return None"""
266  createSymLink = False
267  if logFile:
268  if addDateTime:
269  timeStr = datetime.now().strftime("%d/%m/%Y-%H:%M:%S.%f")
270  text = "{}: {}".format(strftime(timeStr, localtime()).lower(), text)
271 
272  # If the log file is about to be created then we will create a symlink
273  # to the file.
274  if not os.path.isfile(logFile):
275  createSymLink = True
276 
277  with open(logFile, 'a') as fd:
278  if addLF:
279  fd.write("{}\n".format(text))
280  else:
281  fd.write(text)
282 
283  if createSymLink:
284  #This is helpful as the link will point to the latest log file
285  #which can be useful when debugging. I.E no need to find the
286  #name of the latest file.
287  dirName = self._symLinkDir
288  # if the simlink has not been set then default to the logging file
289  if dirName is None:
290  dirName = os.path.dirname(logFile)
291  absSymLink = os.path.join(dirName, symLinkFile)
292  if os.path.lexists(absSymLink):
293  os.remove(absSymLink)
294  os.symlink(logFile, absSymLink)
