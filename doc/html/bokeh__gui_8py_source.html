465 @staticmethod
466 def GetNextUnusedPort(basePort=1024, maxPort = 65534, bindAddress="0.0.0.0"):
467 """@brief Get the first unused above the base port.
468 @param basePort The port to start checking for available ports.
469 @param maxPort The highest port number to check.
470 @param bindAddress The address to bind to.
471 @return The TCP port or -1 if no port is available."""
472 port = basePort
473 while True:
474 try:
475 sock = socket.socket()
476 sock.bind((bindAddress, port))
477 sock.close()
478 break
479 except:
480 port = port + 1
481 if port > maxPort:
482 port = -1
483 break
484
485 return port
