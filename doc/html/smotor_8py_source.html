163  def setAccelDecelFactors(self, accelerationFactor, decelerationFactor, accelerationBinCount=10, decelerationBinCount=10):
164  """@brief Set the factors that determine rate at which the motor accelerates when the motor
165  starts and decelerates as the motor comes to a stop. If both factors are set to
166  1.0 (the maximum) then the motor jumps straight to the required revolutions per
167  second (maximum acceleration) and when the required angle is reached the
168  motor stops immediately. This risks missing/jumping steps particularly if
169  the motor is under load.
170  If the acceleration and deceleration factors are set to 0.5 the motor will
171  accelerate to the required revolutions per second in half the travel time
172  or 0.5 of the required angular movement at which point it will start
173  decelerating at the same rate before finally coming to a stop at the
174  required number of steps.
175  If the acceleration factor + the deceleration factors is greater than 1
176  then the acceleration factor will be followed and then the deceleration
177  factor will be used to slow the motor down. When the required angular
178  movement is reached the motor will stop.
179  @param accelerationFactor The acceleration factor as detailed above.
180  @param decelerationFactor The deceleration factor as detailed above.
181  @param accelerationBinCount The number of bins for each speed change through the acceleration phase (assuming accelerationFactor is not set to 1.0).
182  @param decelerationBinCount The number of bins for each speed change through the deceleration phase (assuming decelerationFactor is not set to 1.0).
183  """
184  self._accelerationFactor = accelerationFactor
185  self._decelerationFactor = decelerationFactor
186  self._accelerationBinCount = accelerationBinCount
187  self._decelerationBinCount = decelerationBinCount

221  def _move(self, stepCount):
222  """@brief Move the motor through the required number of steps.
223  @param stepCount The number of steps to move the motor."""
224  if stepCount > 0:
225  freqHz = self._getFreqHz()
226  micros = int(500000 / freqHz)
227  pulseWave = []
228  pulseWave.append(pigpio.pulse(1 << self._pinout.step, 0, micros)) # pulse on
229  pulseWave.append(pigpio.pulse(0, 1 << self._pinout.step, micros)) # pulse off
230  self._pi.wave_add_generic(pulseWave)
231  waveID = self._pi.wave_create()
232 
233  chainCount = stepCount/SMotor.MAX_WAVE_STEP_COUNT
234  chain = []
235  if chainCount <= 1.0:
236  x = stepCount & 255
237  y = stepCount >> 8
238  chain += [255, 0, waveID, 255, 1, x, y]
239  else:
240  # If more than one chain is required to send all the steps then
241  # send chains with the max number of steps followed by a chain to send the remaining steps (if not 0)
242  stepsSent = 0
243  while stepsSent < stepCount:
244  stepsLeft = stepCount - stepsSent
245  if stepsLeft >= SMotor.MAX_WAVE_STEP_COUNT:
246  #Create a chain with the max number of steps
247  lsb = SMotor.MAX_WAVE_STEP_COUNT&0xff
248  msb = SMotor.MAX_WAVE_STEP_COUNT >> 8
249  chain += [255, 0, waveID, 255, 1, lsb, msb]
250  stepsSent = stepsSent + SMotor.MAX_WAVE_STEP_COUNT
251  else:
252  lsb = stepsLeft&0xff
253  msb = stepsLeft >> 8
254  chain += [255, 0, waveID, 255, 1, lsb, msb]
255  stepsSent = stepsSent + stepsLeft
256 
257  #7 is the number of elements in each chain
258  if (len(chain)/SMotor.ELEMENTS_IN_CHAIN) > SMotor.MAX_CHAIN_COUNT:
259  raise Exception("Unable to move {} steps max = {} steps.".format(stepCount, SMotor.MAX_WAVE_STEP_COUNT*SMotor.MAX_CHAIN_COUNT))
260 
261  self._pi.wave_chain(chain) # Transmit chain.
